package SeleniumFirstTest;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;



public class DayOne 
{

	public static void main(String[] args) throws InterruptedException 
	{
		//open browser
		System.setProperty("webdriver.chrome.driver", "C:/Program Files (x86)/Google/Chrome/Application/chrome.exe");
		WebDriver wd=new ChromeDriver();
		
		//open url
		Thread.sleep(5000);
		wd.get("https://www.google.com/?gws_rd=ssl%22");
		
		//click on gmail
		wd.findElement(By.linkText("Gmail"));
		Thread.sleep(3000);
		
		//enter user name
		wd.findElement(By.id("Email")).sendKeys("g.amoolyareddy@gmail.com");
		Thread.sleep(2000);
		
		//click on next
		wd.findElement(By.id("next")).click();
		Thread.sleep(2000);
		
		//enter password
		wd.findElement(By.id("Passwd")).sendKeys("xyz");
		Thread.sleep(1000);
		
		//click on sign in button
		wd.findElement(By.id("signIn")).click();
		Thread.sleep(3000);
		
		
		
	

	}

}
